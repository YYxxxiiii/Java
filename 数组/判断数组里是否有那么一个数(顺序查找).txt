public class Text{
    public static void main(String[] args) {
        int[] arr = {1, 2, 3, 4};
        System.out.println(find(arr,3));
    }
    public static int find(int[] arr, int toFind){
        for(int i = 0;i < arr.length;i++){
            if(arr[i] == toFind){
                return 1;
            }
        }
        return -1;
    }
}
