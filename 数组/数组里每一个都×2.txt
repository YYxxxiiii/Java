import java.util.Arrays;

public class Text{
    public static void main(String[] args) {
        int[] arr = {1,2,3,4,5};
        mul(arr);
        System.out.println(Arrays.toString(arr));
    }
    public static void mul(int[] arr){
        for(int i = 0;i < arr.length;i++){
            arr[i] = arr[i] * 2;
        }
    }
}
